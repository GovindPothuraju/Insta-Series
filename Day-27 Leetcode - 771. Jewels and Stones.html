Jewels and Stones

Problem Statement:

You are given strings J (types of jewels) and S (stones you have). Count how many stones in S are jewels (characters in J).

Example:

Input: J = "aA", S = "aAAbbbb"

Output: 3

Explanation: Stones 'a', 'A', 'A' are jewels (3 total).

Constraints

  • 1 <= J.length, S.length <= 50
  • Letters in J and S are case-sensitive
  • All characters in J are distinct

Approach 1 ‚Äî Brute Force (Nested Loops)

Time Complexity: O(|J| √ó |S|)
Space Complexity: O(1)

Idea: For every stone, scan all jewels and check equality.

Approach 2 ‚Äî Hash Set (Optimal in Practice)

Time Complexity: O(|J| + |S|) average case
Space Complexity: O(|J|)

Idea: Put J characters in a hash set, then iterate S and test membership.

Approach 3 ‚Äî ASCII Count Array (Fastest)

Time Complexity: O(|J| + |S|)
Space Complexity: O(1) - constant 128/256 bytes

Idea: Use fixed-size array to mark jewels; avoids hash overhead. Best for ASCII input.

Which Approach to Use?

  • Approach 2 (Hash Set): Most real problems, simple and readable, handles Unicode
  • Approach 3 (ASCII Array): Input is known to be ASCII, fastest performance needed
  • Approach 1 (Brute Force): Learning purposes, very small inputs

// Approach 1: Brute Force - O(|J| * |S|)
class Solution {
public:
    int numJewelsInStones(string J, string S) {
        int count = 0;
        // For each stone in S
        for (char s : S) {
            // Check against all jewels
            for (char j : J) {
                if (s == j) {
                    count++;
                    break; // stop checking other jewels
                }
            }
        }
        return count;
    }
};

public class Solution {
    public int numJewelsInStones(String J, String S) {
        int count = 0;
        // Loop through each stone
        for (int i = 0; i < S.length(); i++) {
            char s = S.charAt(i);
            // Check against each jewel
            for (int j = 0; j < J.length(); j++) {
                if (s == J.charAt(j)) {
                    count++;
                    break; // found match, move to next stone
                }
            }
        }
        return count;
    }
}

// Approach 2: Hash Set - O(|J| + |S|)
#include <unordered_set>
class Solution {
public:
    int numJewelsInStones(string J, string S) {
        unordered_set<char> jewels;
        // Store all jewels in hash set
        for (char j : J) jewels.insert(j);
        
        int count = 0;
        // Count stones that are jewels
        for (char s : S) 
            if (jewels.count(s)) 
                count++;
        return count;
    }
};

import java.util.HashSet;
import java.util.Set;

public class Solution {
    public int numJewelsInStones(String J, String S) {
        Set<Character> jewels = new HashSet<>();
        // Convert jewel string to set
        for (char c : J.toCharArray()) 
            jewels.add(c);
        
        int count = 0;
        // Check each stone against jewel set
        for (char s : S.toCharArray()) 
            if (jewels.contains(s)) 
                count++;
        return count;
    }
}

// Approach 3: ASCII Array - O(|J| + |S|)
class Solution {
public:
    int numJewelsInStones(string J, string S) {
        int mark[128] = {0}; // ASCII lookup table
        // Mark all jewels in lookup array
        for (char c : J) 
            mark[(int)c] = 1;
        
        int count = 0;
        // Count using ASCII lookup
        for (char s : S) 
            if (mark[(int)s]) 
                count++;
        return count;
    }
};

public class Solution {
    public int numJewelsInStones(String J, String S) {
        boolean mark[] = new boolean[128]; // ASCII boolean array
        // Set true for all jewel characters
        for (char c : J.toCharArray()) {
            if (c < 128) mark[c] = true;
        }
        
        int count = 0;
        // Count valid jewel stones
        for (char s : S.toCharArray()) {
            if (s < 128 && mark[s]) count++;
        }
        return count;
    }
}
      

# Approach 3: ASCII Array - O(|J| + |S|)
class Solution:
    def numJewelsInStones(self, J: str, S: str) -> int:
        mark = [0] * 128  # ASCII marker array
        # Mark jewel positions
        for c in J:
            if ord(c) < 128:
                mark[ord(c)] = 1
        
        count = 0
        # Count using ASCII lookup
        for s in S:
            if ord(s) < 128 and mark[ord(s)]:
                count += 1
        return count
